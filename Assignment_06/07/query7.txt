MATCH airport = (airportStop:Airport{name: "DAY"})<-[cheapestFlight:CHEAPEST_FLIGHT*1..5]-(airportStart:Airport {name: "LAX"}), 
(airportStop:Airport)<-[:DESTINATION]-(), ()-[:ORIGIN]->(airportStart:Airport ) 
WITH airport, cheapestFlight, reduce(totalPrice=0, rel IN relationships(airport)|totalPrice + rel.price) as cost 
RETURN airport, cost ORDER BY cost LIMIT 1
